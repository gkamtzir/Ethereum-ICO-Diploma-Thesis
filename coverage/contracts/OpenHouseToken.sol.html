        require(_balanceOf[msg.sender] >= numberOfTokens && numberOfTokens > 0);
        Erequire(offer[msg.sender].leasedTo == address(0) && offer[msg.sender].numberOfTokens == 0);

        offer[msg.sender].leasedTimestamp = 0;
